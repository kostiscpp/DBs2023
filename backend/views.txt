-- Create a view for School A
CREATE VIEW view_school_1 AS
SELECT * 
FROM book b
JOIN book_copy bc ON b.ISBN = bc.book_id
WHERE bc.school_id = 1;

-- Create a view for School B
CREATE VIEW view_school_2 AS
SELECT * 
FROM book b
JOIN book_copy bc ON b.ISBN = bc.book_id
WHERE bc.school_id = 2;
