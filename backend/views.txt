-- Create a view for School A
CREATE VIEW school_a_books_view AS
SELECT b.title, b.ISBN, b.edition
FROM book b
JOIN book_copy bc ON b.ISBN = bc.ISBN
WHERE bc.school_id = <school_a_id>;

-- Create a view for School B
CREATE VIEW school_b_books_view AS
SELECT b.title, b.ISBN, b.edition
FROM book b
JOIN book_copy bc ON b.ISBN = bc.ISBN
WHERE bc.school_id = <school_b_id>;
